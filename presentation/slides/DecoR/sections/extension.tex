\section{Extension: Dense Confounding}

\begin{frame}{The "Dense $Y$" Challenge}
    \begin{alertblock}{What if the sparse assumption fails?}
        The authors hypothesize about a scenario where the confounder $U$ has a \textbf{sparse} effect on $X$, but is \textbf{dense} toward $Y$.
    \end{alertblock}
    
    \vspace{0.2cm}
    \textbf{Why Standard DeCoR Fails Here:}
    \begin{itemize}
        \item $U$ pollutes every frequency in $Y$.
        \item Standard Torrent looks for large residuals: $r_k = |Y_k - X_k^\top \hat{\beta}|$.
        \item Because $U$ is everywhere, $r_k$ is huge everywhere. Torrent gets confused and throws away clean data.
    \end{itemize}
\end{frame}

\begin{frame}{The Proposed Solution: Covariance Thresholding}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{The Minor Modification:} \\
            Instead of thresholding based on the residual alone, we threshold based on the \textbf{covariance}:
            \begin{equation*}
                v_k = |X_k (Y_k - X_k^\top \hat{\beta})|
            \end{equation*}
        \end{column}
        \begin{column}{0.5\textwidth}
            \begin{exampleblock}{Why it works}
                By asymptotic orthogonality, the independent noise in $X$ and the dense noise in $U$ cancel out at the clean frequencies:
                $$ (T_k^{\phi,n}(X))_{G_n^c}^\top (T_k^{\phi,n}(U))_{G_n^c} \to 0 $$
            \end{exampleblock}
        \end{column}
    \end{columns}
\end{frame}
