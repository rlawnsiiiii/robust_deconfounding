        We set in our one-dimensional scenario:
    \begin{itemize}
        \item $U := \text{AR(1)} + \epsilon_u$ with AR(1) $:=$ Ornstein-Uhlenbeck process and noise $\epsilon_u \sim N(0,1)$
        \item $U_{sparse} = \text{sparsify($U$, $|G_n| = 0.25 \cdot n$, cosine-basis)}$
        \item $X := \text{AR(2)} + U_{sparse} + \epsilon_x$ with $\epsilon_x \sim N(0,1)$
        \item $Y := 3X + 10 \cdot U_{spasre} + \epsilon_y$ with $\epsilon_y \sim N(0,1)$
    \end{itemize}
\end{frame}
