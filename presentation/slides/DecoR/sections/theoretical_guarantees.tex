\section{Theoretical Guarantees}

\begin{frame}[shrink=10]{Setting the Stage for Proofs for Torrent}
    \begin{block}{Key Assumptions}
        \begin{enumerate}
            \item \textbf{Setting 2:} Let $n \geq d$. We observe $X \in \mathbb{R}^{n \times d}$ and $Y \in \mathbb{R}^n$ defined by:
        \begin{equation*}
            Y := X\beta + \epsilon + o
        \end{equation*}
            where $\epsilon \in \mathbb{R}^n$ is noise and $o \in \mathbb{R}^n$ are sparse outliers supported on $G_n$ (i.e., $o_i = 0$ for $i \notin G_n$). We call $G_n^c$ the \textbf{inliers}.
            \item \textbf{Bounded Outliers $G_n$:} \# confounded frequencies $|G_n| \le c_n$.
            \item \textbf{Mistakes-Stability-Ratio:} As $n \to \infty$, it holds with high probability that:
            $$ \eta := \max_{S \text{ s.t. } |S|=n-c_n} \frac{||X_{V(S)}||_2}{\sqrt{\lambda_{\min}(X_S^\top X_S)}} < \frac{1}{\sqrt{2}} $$
            where $S \subseteq \{1, ..., n\}$ and $V(S) = (S \cup G_n^c) \setminus (G_n^c \cap S)$ is the symmetric difference between $S$ and $G_n^c$.
        \end{enumerate}
    \end{block}

    \textbf{Intuitions:}
    \begin{itemize}
        \item Should $\eta$ better be nearly $0$ or close to our upper bound?
        \item What happens if $X_S$ is singular for some $S$?
    \end{itemize}
\end{frame}

\begin{frame}{Proofs - Torrent (deterministic)}
    \begin{block}{Guarantee of Torrent (Theorem 3)}
    Let $S_t$ be the final estimated subset of Torrent on data $X$, $Y$ using threshold parameter $a_n := n - c_n$. Then
        \begin{equation*}
        \begin{aligned}
        \|\hat{\beta}_{\text{Tor}}^{n,a_n} - \beta\|_2 &\leq \text{Err}_{\text{OLS}} \\
        &\quad + \frac{\sqrt{2}\|X_{V(S_t)}\|_2 \cdot \text{Err}_{\text{OLS}} + \sqrt{2}\|\epsilon_{V(S_t)}\|_2}{\sqrt{\lambda_{\min}(X_{S_t}^\top X_{S_t})}(1 - \sqrt{2}\eta)}.
        \end{aligned}
        \end{equation*}
        
        \small with $\text{Err}_{\text{OLS}} = \|(X_{S_t}^\top X_{S_t})^{-1} X_{S_t}^\top \epsilon_{S_t}\|_2$ \textit{being the baseline error of the clean subset.}
    \end{block}
    \textbf{Intuitions:}
    \begin{itemize}
        \item What happens in case of no outliers at all?
        \item What happens if $\eta \approx \frac{1}{\sqrt{2}}$ or $X_{S_t}$ is singular?
    \end{itemize}
\end{frame}

\begin{frame}[shrink=15]{\normalsize Proofs - DecoR-Tor}
    \begin{block}{Convergence properties of \texttt{DecoR-Tor} (Theorem 6)}
        Let $c_n$ be a known sequence of non-negative natural numbers such that $|G_n| \leq c_n$ and let Assumption 2 in \cite{bhatia2015robustregressionhardthresholding} be satisfied. Assume that for all $\delta > 0$ there exists $\bar{n} \in \mathbb{N}$ such that for all $n \geq \bar{n}$ it holds that
        \begin{equation*}
            \mathbb{P}\left[ \max_{S \subseteq \{1,\dots,n\} \text{ s.t. } |S|=n-c_n} \frac{\|(X_{\phi}^n)_{V(S)}\|_2}{\sqrt{\lambda_{\min}\left((X_{\phi}^n)_S^\top (X_{\phi}^n)_S\right)}} < \frac{1}{\sqrt{2}} \right] \geq 1 - \delta.
        \end{equation*}
        If \texttt{DecoR} is executed with \texttt{Torrent} and the sequence of threshold parameters $a_n = n - c_n$, then
        \begin{equation*}
            \|\hat{\beta}_{\text{DecoR}}^{\phi,n} - \beta\|_2 \in \mathcal{O}_{\mathbb{P}}\left( \sigma \sqrt{\frac{c_n \log(n/c_n)}{n}} \right).
        \end{equation*}
    \end{block}
    \textbf{Insights:}
    \begin{itemize}
        \item $\hat{\beta}_{\text{DecoR}}^{\phi,n}$ is a consistent estimator of $\beta$, if $\sigma=0$ or $c_n \in o(n)$
        \item if noise is non-zero and $c_n$ is asymptotically proportional to $n$, then the estimation error is bounded by $\sigma$, up to constant factors.
    \end{itemize}
\end{frame}
